    function updateRGB()
        rgbTimer = rgbTimer + 1

        if rgbTimer > 0 and rgbTimer <= 100 then
            r = 255

            g = rgbTimer * 2.55

            b = 0
        elseif rgbTimer > 100 and rgbTimer <= 200 then
            r = 255 - (rgbTimer - 100) * 2.55

            g = 255

            b = 0
        elseif rgbTimer > 200 and rgbTimer <= 300 then
            r = 0

            g = 255

            b = (rgbTimer - 200) * 2.55
        elseif rgbTimer > 300 and rgbTimer <= 400 then
            r = 0

            g = 255 - (rgbTimer - 300) * 2.55

            b = 255
        elseif rgbTimer > 400 and rgbTimer <= 500 then
            r = (rgbTimer - 400) * 2.55

            g = 0

            b = 255
        elseif rgbTimer > 500 and rgbTimer <= 600 then
            r = 255

            g = 0

            b = 255 - (rgbTimer - 500) * 2.55
        end

        if rgbTimer > 600 then
            rgbTimer = 0
        end
    end
